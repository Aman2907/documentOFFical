let nums= [1,2,2,3, 5,6,7,7,8,9], x = 2

function linearSrc(nums, x){
    let start = 0;
    let end = nums.length - 1;
    let ans = nums.length;
    
    while(start <= end){
        let mid = Math.floor((start + end) / 2);
        
        if(nums[mid] >= x){
            ans = mid;
            end = mid - 1;
        }else{
            start = mid + 1;
        }
    }
    return ans;
}

console.log(linearSrc(nums, x))

O(1) space
