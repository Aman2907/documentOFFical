let nums1 = [1,2,2,1]
let nums2 = [2,2]
// Output: [2]

function InterSection(nums1, nums2){
    let set1 = new Set(nums1);
    let result = new Set();
    
    for(let i = 0 ; i < nums2.length; i++) {
        if(set1.has(nums2[i])){
            result.add(nums2[i]);
        }
    }
    return Array.from(result)
}
console.log(InterSection(nums1, nums2))
Time Complexity: O(n + m)
Space Complexity: O(n + m)
