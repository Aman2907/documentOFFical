let target = 7
let nums = [2,3,1,2,4,3]

function subArr(target, nums){
    let minLen = Infinity;
    for(let i = 0 ; i < nums.length; i++){
        let sum = 0;
        for(let j = i ; j < nums.length ; j++){
            sum = sum + nums[j]
           if(sum >= target){
               minLen = Math.min(minLen, j - i + 1)
           }
        }
    }
    return minLen === Infinity ? 0 : minLen;
}
console.log(subArr(target, nums))
