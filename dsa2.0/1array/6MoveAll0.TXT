function moveZero(arr){
    
    let temp = [];
    for(let i = 0 ; i < arr.length ; i++){
        if(arr[i] !== 0) {
            temp.push(arr[i])
        }
    }
    
    let z = temp.length;

    for(let i = 0 ; i < z ; i++){
        arr[i] = temp[i];
    }

    for(let i = z; i < arr.length; i++){
        arr[i] = 0;
    }
    return arr;
   
}
